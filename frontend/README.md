# Docker Commands

# Build
docker build -t frontend .

# Run
docker run --name frontend -p 3000:3000 --restart=always -d frontend

# Kill
docker kill frontend

# Remove Container (only after Kill)
docker rm frontend

# Remove image
docker image rm frontend

# Save image as TAR
docker image save --output frontend.tar frontend

# Load image as TAR
docker load --input frontend.tar
